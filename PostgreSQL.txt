-- Crear base de datos
CREATE DATABASE "LoginDioris";

-- Cambiar a la base de datos
\c "LoginDioris";

-- Tabla: Registro cliente
CREATE TABLE "Registro cliente" (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100),
    correo VARCHAR(100),
    telefono VARCHAR(20)
);
----------
-- Tabla: Login
CREATE TABLE Login (
    id SERIAL PRIMARY KEY,
    usuario VARCHAR(50),
    clave VARCHAR(100),
    compañia VARCHAR(100)
);
----------
-- Tabla: registro
CREATE TABLE registro (
    id SERIAL PRIMARY KEY,
    usuario VARCHAR(50),
    telefono VARCHAR(20),
    "E-mail" VARCHAR(100),
    contraseña VARCHAR(100)
);
