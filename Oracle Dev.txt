-- Nota: En Oracle no se crea la base de datos desde SQL, solo trabajamos dentro del esquema.

-- Tabla: Registro cliente
CREATE TABLE "Registro cliente" (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR2(100),
    correo VARCHAR2(100),
    telefono VARCHAR2(20)
);
----------
-- Tabla: Login
CREATE TABLE Login (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    usuario VARCHAR2(50),
    clave VARCHAR2(100),
    compañia VARCHAR2(100)
);
----------
-- Tabla: registro
CREATE TABLE registro (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    usuario VARCHAR2(50),
    telefono VARCHAR2(20),
    "E-mail" VARCHAR2(100),
    contraseña VARCHAR2(100)
);
